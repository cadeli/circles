More complex movements

var starpath  = new Starpath(document.getElementById('my_div'))
.addSprite(new Sprite().setSprite('../assets/gemBlue.png')
.setCenter(0,0).setSpeed(0.5).setPhase(1/2)
.setAmplitudeExt(1,1,0,0,1/8,1/8,0,0))
			
.addSprite(new Sprite().setSprite('../assets/coinGold.png')
.setCenter(0,0).setSpeed(0.5).setPhase(1)
.setAmplitudeExt(1,1,0,0,1/8,1/8,0,0))
animate(starpath);
