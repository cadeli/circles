Example 3

Create 3 stars with zoom and position variations

var starpath  = new Starpath(document.getElementById('my_div'))
.addSprite(new Sprite().setSprite('../assets/star.png').setCenter(0,0).setZoom(2,2))
.addSprite(new Sprite().setSprite('../assets/star.png').setCenter(-0.5,0).setAmplitude(0,0))
.addSprite(new Sprite().setSprite('../assets/star.png').setCenter(0.5,0).setAmplitude(0,0))
animate(starpath);
